17  package org.atmosphere.sockjs;
18  
19  import com.fasterxml.jackson.core.JsonGenerationException;
20  import com.fasterxml.jackson.core.JsonGenerator;
21  import com.fasterxml.jackson.core.JsonProcessingException;
22  import com.fasterxml.jackson.core.io.CharTypes;
23  import com.fasterxml.jackson.core.json.JsonWriteContext;
24  import com.fasterxml.jackson.databind.JsonSerializer;
25  import com.fasterxml.jackson.databind.ObjectMapper;
26  import com.fasterxml.jackson.databind.SerializerProvider;
27  import com.fasterxml.jackson.databind.module.SimpleModule;
28  
29  import java.io.IOException;
30  
31  /**
32   * SockJS requires a special JSON codec - it requires that many other characters,
33   * over and above what is required by the JSON spec are escaped.
34   * To satisfy this we escape any character that escapable with short escapes and
35   * any other non ASCII character we unicode escape it
36   *
37   * @author <a href="http://tfox.org">Tim Fox</a>
38   */
39  public class JsonCodec {
40  
41      private final static ObjectMapper mapper;
42  
43      static {
44          mapper = new ObjectMapper();
45  
46          // By default, Jackson does not escape unicode characters in JSON strings
47          // This should be ok, since a valid JSON string can contain unescaped JSON
48          // characters.
49          // However, SockJS requires that many unicode chars are escaped. This may
50          // be due to browsers barfing over certain unescaped characters
51          // So... when encoding strings we make sure all unicode chars are escaped
52  
53          // This code adapted from http://wiki.fasterxml.com/JacksonSampleQuoteChars
54          SimpleModule simpleModule = new SimpleModule();
55  
56          simpleModule.addSerializer(String.class, new JsonSerializer<String>() {
57              final char[] HEX_CHARS = "0123456789abcdef".toCharArray();
58              final int[] ESCAPE_CODES = CharTypes.get7BitOutputEscapes();
59  
60              private void writeUnicodeEscape(JsonGenerator gen, char c) throws IOException {
61                  gen.writeRaw('\\');
62                  gen.writeRaw('u');
63                  gen.writeRaw(HEX_CHARS[(c >> 12) & 0xF]);
64                  gen.writeRaw(HEX_CHARS[(c >> 8) & 0xF]);
65                  gen.writeRaw(HEX_CHARS[(c >> 4) & 0xF]);
66                  gen.writeRaw(HEX_CHARS[c & 0xF]);
67              }
68  
69              private void writeShortEscape(JsonGenerator gen, char c) throws IOException {
70                  gen.writeRaw('\\');
71                  gen.writeRaw(c);
72              }
73  
74              @Override
75              public void serialize(String str, JsonGenerator gen, SerializerProvider provider) throws IOException {
76                  int status = ((JsonWriteContext) gen.getOutputContext()).writeValue();
77                  switch (status) {
78                      case JsonWriteContext.STATUS_OK_AFTER_COLON:
79                          gen.writeRaw(':');
80                          break;
81                      case JsonWriteContext.STATUS_OK_AFTER_COMMA:
82                          gen.writeRaw(',');
83                          break;
84                      case JsonWriteContext.STATUS_EXPECT_NAME:
85                          throw new JsonGenerationException("Can not write string value here");
86                  }
87                  gen.writeRaw('"');
88                  for (char c : str.toCharArray()) {
89                      if (c >= 0x80) writeUnicodeEscape(gen, c); // use generic escaping for all non US-ASCII characters
90                      else {
91                          // use escape table for first 128 characters
92                          int code = (c < ESCAPE_CODES.length ? ESCAPE_CODES[c] : 0);
93                          if (code == 0) gen.writeRaw(c); // no escaping
94                          else if (code == -1) writeUnicodeEscape(gen, c); // generic escaping
95                          else writeShortEscape(gen, (char) code); // short escaping (\n \t ...)
96                      }
97                  }
98                  gen.writeRaw('"');
99              }
100         });
101         mapper.registerModule(simpleModule);
102     }
103 
104     public static String encode(Object obj) {
105         try {
106             return mapper.writeValueAsString(obj);
107         } catch (JsonProcessingException e) {
108             return null;
109         }
110     }
111 
112     public static Object decodeValue(String str, Class<?> clazz) {
113         try {
114             return mapper.readValue(str, clazz);
115         } catch (IOException e) {
116             return null;
117         }
118     }
