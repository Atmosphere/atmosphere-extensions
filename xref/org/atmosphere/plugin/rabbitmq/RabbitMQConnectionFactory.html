16  package org.atmosphere.plugin.rabbitmq;
17  
18  import com.rabbitmq.client.Channel;
19  import com.rabbitmq.client.Connection;
20  import com.rabbitmq.client.ConnectionFactory;
21  import org.atmosphere.cpr.AtmosphereConfig;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.io.IOException;
26  
27  /**
28   * RabbitMQ Connection Factory.
29   *
30   * @author Thibault Normand
31   * @author Jean-Francois Arcand
32   */
33  public class RabbitMQConnectionFactory implements AtmosphereConfig.ShutdownHook{
34      private static final Logger logger = LoggerFactory.getLogger(RabbitMQBroadcaster.class);
35  
36      private static RabbitMQConnectionFactory factory;
37  
38      public static final String PARAM_HOST = RabbitMQBroadcaster.class.getName() + ".host";
39      public static final String PARAM_USER = RabbitMQBroadcaster.class.getName() + ".user";
40      public static final String PARAM_PASS = RabbitMQBroadcaster.class.getName() + ".password";
41      public static final String PARAM_EXCHANGE_TYPE = RabbitMQBroadcaster.class.getName() + ".exchange";
42      public static final String PARAM_VHOST = RabbitMQBroadcaster.class.getName() + ".vhost";
43      public static final String PARAM_PORT = RabbitMQBroadcaster.class.getName() + ".port";
44  
45      private String exchangeName;
46      private ConnectionFactory connectionFactory;
47      private Connection connection;
48      private Channel channel;
49      private String exchange;
50  
51      public RabbitMQConnectionFactory(AtmosphereConfig config) {
52  
53          String s = config.getInitParameter(PARAM_EXCHANGE_TYPE);
54          if (s != null) {
55              exchange = s;
56          } else {
57              exchange = "topic";
58          }
59  
60          String host = config.getInitParameter(PARAM_HOST);
61          if (host == null) {
62              host = "127.0.0.1";
63          }
64  
65          String vhost = config.getInitParameter(PARAM_VHOST);
66          if (vhost == null) {
67              vhost = "/";
68          }
69  
70          String user = config.getInitParameter(PARAM_USER);
71          if (user == null) {
72              user = "guest";
73          }
74  
75          String port = config.getInitParameter(PARAM_PORT);
76          if (port == null) {
77              port = "5672";
78          }
79  
80          String password = config.getInitParameter(PARAM_PASS);
81          if (password == null) {
82              password = "guest";
83          }
84  
85          exchangeName = "atmosphere." + exchange;
86          try {
87              logger.debug("Create Connection Factory");
88              connectionFactory = new ConnectionFactory();
89              connectionFactory.setUsername(user);
90              connectionFactory.setPassword(password);
91              connectionFactory.setVirtualHost(vhost);
92              connectionFactory.setHost(host);
93              connectionFactory.setPort(Integer.valueOf(port));
94  
95              logger.debug("Try to acquire a connection ...");
96              connection = connectionFactory.newConnection();
97              channel = connection.createChannel();
98  
99              logger.debug("Topic creation '{}'...", exchangeName);
100             channel.exchangeDeclare(exchangeName, exchange);
101         } catch (Exception e) {
102             String msg = "Unable to configure RabbitMQBroadcaster";
103             logger.error(msg, e);
104             throw new RuntimeException(msg, e);
105         }
106         config.shutdownHook(this);
107     }
108 
109     public final static RabbitMQConnectionFactory getFactory(AtmosphereConfig config) {
110         // No need to synchronize here as the first Broadcaster created is at startup.
111         if (factory == null) {
112             factory = new RabbitMQConnectionFactory(config);
113         }
114         return factory;
115     }
116 
117     public String exchangeName() {
118         return exchangeName;
119     }
120 
121     public Channel channel() {
122         return channel;
123     }
124 
125     @Override
126     public void shutdown() {
127         try {
128             channel.close();
129             connection.close();
130         } catch (IOException e) {
131             logger.trace("", e);
132         }
133     }
134 }
