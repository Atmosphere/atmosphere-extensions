61  public class SockJsAtmosphereInterceptor extends AtmosphereInterceptorAdapter {
62  
63      public final static String SOCKS_JS_ORIGIN = SockJsAtmosphereInterceptor.class.getName() + ".origin";
64  
65      private static final Logger logger = LoggerFactory.getLogger(SockJsAtmosphereInterceptor.class);
66      private boolean supportWebSocket = true;
67      private final AtomicReference<String> baseURL = new AtomicReference<String>("");
68      private AtmosphereFramework framework;
69      private final Map<String, SockjsSession> sessions = Collections.synchronizedMap(new WeakHashMap<String, SockjsSession>());
70  
71      public final static AtmosphereHandler ECHO_ATMOSPHEREHANDLER = new AbstractReflectorAtmosphereHandler() {
72          @Override
73          public void onRequest(AtmosphereResource resource) throws IOException {
74              String body = IOUtils.readEntirely(resource).toString();
75              if (!body.isEmpty()) {
76                  resource.getBroadcaster().broadcast(body);
77              }
78          }
79      };

81      @Override
82      public void configure(final AtmosphereConfig config) {
83          framework = config.framework();
84          supportWebSocket = config.framework().getAsyncSupport().supportWebSocket();
85          config.properties().put(HeaderConfig.JSONP_CALLBACK_NAME, "c");
86          for (AtmosphereInterceptor i : framework.interceptors()) {
87              if (HeartbeatInterceptor.class.isAssignableFrom(i.getClass())) {
88                  HeartbeatInterceptor.class.cast(i).paddingText("h".getBytes()).heartbeatFrequencyInSeconds(25);
89              }
90          }
91  
92          if (config.handlers().size() == 0) {
93              framework.addAtmosphereHandler("/*", ECHO_ATMOSPHEREHANDLER);
94          }
95      }

97      @Override
98      public Action inspect(final AtmosphereResource r) {
99          final AtmosphereRequest request = r.getRequest();
100 
101         if (request.getAttribute("sockjs.skipInterceptor") != null) {
102             return Action.CONTINUE;
103         }
104 
105         boolean info = request.getRequestURI().endsWith("/info");
106         if (info) {
107             return info(r);
108         }
109 
110         boolean iframe = request.getRequestURI().endsWith("/iframe.html");
111         if (iframe) {
112             return iframe(r);
113         }
114 
115         if (!baseURL.get().isEmpty() && request.getRequestURI().startsWith(baseURL.get())) {
116             super.inspect(r);
117 
118             final String sessionId = param(request.getRequestURI(), 2);
119             String transport = param(request.getRequestURI(), 4);
120 
121             SockjsSession s = sessions.get(sessionId);
122             configureTransport(AtmosphereResourceImpl.class.cast(r), transport, s != null);
123             boolean longPolling = org.atmosphere.util.Utils.resumableTransport(r.transport());
124 
125             if (s == null) {
126                 sessions.put(sessionId, new SockjsSession());
127                 if (!longPolling) {
128                     installWriter(r, sessionId);
129                 }
130                 return Action.CONTINUE;
131             } else if (longPolling) {
132                 installWriter(r, sessionId);
133                 return Action.CONTINUE;
134             }
135 
136             return injectMessage(r);
137         }
138         return Action.CONTINUE;
139     }

141     private Action iframe(AtmosphereResource r) {
142         final AtmosphereResponse response = r.getResponse();
143         response.setContentType("text/html");
144         String origin = framework.getAtmosphereConfig().getInitParameter(SOCKS_JS_ORIGIN);
145         if (origin == null) {
146             origin = "http://localhost:8080/lib/sockjs.js";
147         }
148         try {
149             response.write(IFrameUtils.generateIFrame(origin)).flushBuffer();
150         } catch (IOException e) {
151             logger.error("", e);
152         }
153         return Action.CANCELLED;
154     }
155 
156     protected Action info(AtmosphereResource r) {
157         final AtmosphereResponse response = r.getResponse();
158         final AtmosphereRequest request = r.getRequest();
159 
160         response.headers().put("Content-Type", "application/json; charset=UTF-8");
161         ObjectNode json = new ObjectNode(JsonNodeFactory.instance);
162         json.put("websocket", supportWebSocket);
163         json.putArray("origins").add("*:*");
164         json.put("entropy", new Random().nextInt());
165         r.write(JsonCodec.encode(json));
166 
167         if (baseURL.get().isEmpty()) {
168             baseURL.set(request.getRequestURI().substring(0, request.getRequestURI().indexOf("/info")));
169         }
170 
171         return Action.CANCELLED;
172     }
173 
174     private String param(String url, int pos) {
175         String[] params = url.split("/");
176         if (params.length < pos) {
177             return "/";
178         } else {
179             return params[pos];
180         }
181 
182     }

184     private void configureTransport(AtmosphereResourceImpl r, String s, boolean hasSession) {
185         if ("websocket".equals(s)) {
186             r.transport(WEBSOCKET).addEventListener(new WebSocketTransport());
187         } else if ("xhr".equals(s) || "xdr".equals(s)) {
188             r.transport(LONG_POLLING);
189 
190             if (!hasSession) {
191                 r.addEventListener(new LongPollingTransport());
192             }
193         } else if ("xhr_streaming".equals(s)) {
194             r.transport(STREAMING).addEventListener(new StreamingTransport());
195         } else if ("jsonp".equals(s)) {
196             r.transport(JSONP);
197 
198             if (!hasSession) {
199                 r.addEventListener(new JSONPTransport());
200             }
201         } else if ("eventsource".equals(s)) {
202             r.transport(SSE).addEventListener(new SSETransport());
203         } else if (s.indexOf("_send") != -1) {
204             r.transport(POLLING);
205         } else if ("htmlfile".equals(s)) {
206             r.transport(HTMLFILE).addEventListener(new HtmlFileTransport());
207         } else if (s.indexOf("_send") != -1) {
208         } else {
209             r.transport(UNDEFINED).addEventListener(new StreamingTransport());
210         }
211     }

213     private void installWriter(final AtmosphereResource r, final String sessionId) {
214         final AtmosphereResource.TRANSPORT transport = r.transport();
215         final AtmosphereResponse response = r.getResponse();
216 
217         AsyncIOWriter writer = response.getAsyncIOWriter();
218         if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
219             AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
220                 @Override
221                 public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
222                     String charEncoding = response.getCharacterEncoding() == null ? "UTF-8" : response.getCharacterEncoding();
223                     String s = new String(responseDraft, charEncoding);
224 
225                     // Ugly.
226                     if (s.equalsIgnoreCase("h") || s.equals("c") || (s.equals("o\n") && r.transport().equals(AtmosphereResource.TRANSPORT.WEBSOCKET))) {
227                         return s.getBytes();
228                     }
229 
230                     if (!s.isEmpty()) {
231                         try {
232                             if (transport.equals(JSONP)) {
233                                 return ("a" + s).getBytes(charEncoding);
234                             } else if (transport.equals(HTMLFILE)) {
235                                 StringBuilder sb = new StringBuilder();
236                                 sb.append("<script>\np(")
237                                         .append("\"")
238                                         .append(StringEscapeUtils.escapeJavaScript("a[\"" + StringEscapeUtils.escapeJavaScript(s) + "\"]\n"))
239                                         .append("\")</script>\n");
240                                 return (sb.toString()).getBytes(charEncoding);
241                             } else {
242                                 return ("a[\"" + StringEscapeUtils.escapeJavaScript(s) + "\"]\n").getBytes(charEncoding);
243 
244                             }
245                         } catch (Exception e) {
246                             logger.error("", e);
247                             return "".getBytes();
248                         }
249                     }
250 
251                     return s.getBytes();
252                 }
253             });
254         } else {
255             logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
256         }
257 
258         r.addEventListener(new AtmosphereResourceEventListenerAdapter() {
259             @Override
260             public void onDisconnect(AtmosphereResourceEvent event) {
261                 sessions.remove(sessionId);
262             }
263         });
264     }

266     private Action injectMessage(AtmosphereResource r) {
267         final AtmosphereResponse response = r.getResponse();
268         final AtmosphereRequest request = r.getRequest();
269 
270         try {
271             String body = IOUtils.readEntirely(r).toString();
272             if (!body.isEmpty() && body.startsWith("d=")) {
273                 body = URLDecoder.decode(body, "UTF-8");
274                 body = body.substring(2);
275                 response.setStatus(200);
276                 response.write("ok", true).flushBuffer();
277                 reInject(request, response, body);
278             } else {
279                 String[] messages = parseMessageString(body);
280                 for (String m : messages) {
281                     if (m == null) continue;
282                     reInject(request, response, m);
283                 }
284                 response.setStatus(204);
285             }
286         } catch (Exception e) {
287             logger.error("", e);
288         }
289         return Action.CANCELLED;
290     }

292     private void reInject(AtmosphereRequest request, AtmosphereResponse response, String body) throws IOException, ServletException {
293         request.setAttribute("sockjs.skipInterceptor", Boolean.TRUE);
294         framework.doCometSupport(request.body(body), response);
295         request.setAttribute("sockjs.skipInterceptor", null);
296     }
297 
298     private String[] parseMessageString(String msgs) {
299         try {
300             String[] parts;
301             if (msgs.startsWith("[")) {
302                 //JSON array
303                 parts = (String[]) JsonCodec.decodeValue(msgs, String[].class);
304             } else {
305                 //JSON string
306                 String str = (String) JsonCodec.decodeValue(msgs, String.class);
307                 parts = new String[]{str};
308             }
309             return parts;
310         } catch (Exception e) {
311             return null;
312         }
313     }
