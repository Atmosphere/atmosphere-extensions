18  import java.io.UnsupportedEncodingException;
19  
20  public class IFrameUtils {
21  
22      public static byte[] generateIFrame(String origin) {
23          StringBuilder b = new StringBuilder();
24          b.append("<!DOCTYPE html>\n").append(
25                  "<html>\n").append(
26                  "<head>\n").append(
27                  "  <meta http-equiv=\"X-UA-Compatible\" content=\"IE=edge\" />\n").append(
28                  "  <meta http-equiv=\"Content-Type\" content=\"text/html; charset=UTF-8\" />\n").append(
29                  "  <script>\n").append(
30                  "    document.domain = document.domain;\n").append(
31                  "    _sockjs_onload = function(){SockJS.bootstrap_iframe();};\n").append(
32                  "  </script>\n").append(
33                  "  <script src=\"").append(origin).append("\"></script>\n").append(
34                  "</head>\n").append(
35                  "<body>\n").append(
36                  "  <h2>Don't panic!</h2>\n").append(
37                  "  <p>This is a SockJS hidden iframe. It's used for cross domain magic.</p>\n").append(
38                  "</body>\n").append(
39                  "</html>");
40          try {
41              return b.toString().getBytes("UTF-8");
42          } catch (UnsupportedEncodingException e) {
43              return b.toString().getBytes();
44          }
45      }

47      public static byte[] generateHtmlFile(String callback) {
48          StringBuilder b = new StringBuilder();
49          b.append("<!doctype html>\n").append(
50                  "<html><head>\n").append(
51                  "  <meta http-equiv=\"X-UA-Compatible\" content=\"IE=edge\" />\n").append(
52                  "  <meta http-equiv=\"Content-Type\" content=\"text/html; charset=UTF-8\" />\n").append(
53                  "</head><body><h2>Don't panic!</h2>\n").append(
54                  "  <script>\n").append(
55                  "    document.domain = document.domain;\n").append(
56                  "    var c = parent.").append(callback).append(";\n").append(
57                  "    c.start();\n").append(
58                  "    function p(d) {c.message(d);};\n").append(
59                  "    window.onload = function() {c.stop();};\n").append(
60                  "  </script>");
61  
62          try {
63              return b.toString().getBytes("UTF-8");
64          } catch (UnsupportedEncodingException e) {
65              return b.toString().getBytes();
66          }

68      }
