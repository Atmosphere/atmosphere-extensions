29      public enum State {
30          MESSAGE_RECEIVED,
31          MESSAGE_PUBLISHED,
32          OPENING,
33          RE_OPENING,
34          CLOSED,
35          ERROR;
36  
37          @Override
38          public String toString() {
39              switch (this) {
40                  case MESSAGE_RECEIVED:
41                      return "messageReceived";
42                  case MESSAGE_PUBLISHED:
43                      return "messagePublished";
44                  case OPENING:
45                      return "opening";
46                  case RE_OPENING:
47                      return "re-opening";
48                  case CLOSED:
49                      return "closed";
50                  default:
51                  case ERROR:
52                      return "error";
53              }
54          }
55  
56          public static State fromString(String s) {
57              for (State st : State.values()) {
58                  if (st.toString().equals(s)) {
59                      return st;
60                  }
61              }
62              return State.ERROR;
63          }
64      }

85      @Override
86      public <T> List<T> getMessages() {
87          Object containedMessage = getMessageObject();
88          if (containedMessage == null) {
89              return Collections.emptyList();
90          } else if (containedMessage instanceof List) {
91              return (List) containedMessage;
92          } else {
93              return (List<T>) Collections.singletonList(containedMessage);
94          }
95      }
