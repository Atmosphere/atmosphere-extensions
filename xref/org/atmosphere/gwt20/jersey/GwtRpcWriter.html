9   import java.io.IOException;
10  import java.io.OutputStream;
11  import java.lang.annotation.Annotation;
12  import java.lang.reflect.Type;
13  import java.util.Collections;
14  import java.util.List;
15  import javax.ws.rs.Produces;
16  import javax.ws.rs.WebApplicationException;
17  import javax.ws.rs.core.MediaType;
18  import javax.ws.rs.core.MultivaluedMap;
19  import javax.ws.rs.ext.MessageBodyWriter;
20  import javax.ws.rs.ext.Provider;
21  
22  import org.atmosphere.gwt20.shared.Constants;
23  import org.atmosphere.gwt20.server.GwtRpcUtil;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * @author p.havelaar
29   */
30  @Provider
31  @Produces(Constants.GWT_RPC_MEDIA_TYPE)
32  public class GwtRpcWriter implements MessageBodyWriter<Object> {
33  
34      private final static Logger logger = LoggerFactory.getLogger(GwtRpcWriter.class);
35  
36      @Override
37      public boolean isWriteable(Class<?> type, Type type1, Annotation[] antns, MediaType mt) {
38          return mt.isCompatible(MediaTypes.GWT_RPC_MEDIA_TYPE);
39      }
40  
41      @Override
42      public long getSize(Object t, Class<?> type, Type type1, Annotation[] antns, MediaType mt) {
43          return -1;
44      }
45  
46      @Override
47      public void writeTo(Object t, Class<?> type, Type type1, Annotation[] antns, MediaType mt, MultivaluedMap<String, Object> headers, OutputStream out) throws IOException, WebApplicationException {
48          try {
49              List ct = Collections.singletonList(Constants.GWT_RPC_MEDIA_TYPE + "; charset=UTF-8");
50              if (!ct.equals(headers.get("Content-Type"))) {
51                  headers.put("Content-Type", ct);
52              }
53              if (t instanceof String && ((String) t).isEmpty()) {
54                  out.flush();
55                  return;
56              }
57              String data = GwtRpcUtil.serialize(t);
58              out.write(data.getBytes("UTF-8"));
59              out.flush();
60          } catch (SerializationException ex) {
61              logger.error("Failed to serialize message", ex);
62          }
63      }
64  
65  }
