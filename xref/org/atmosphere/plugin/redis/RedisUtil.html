16  package org.atmosphere.plugin.redis;
17  
18  import org.apache.commons.pool.impl.GenericObjectPool;
19  import org.atmosphere.cpr.AtmosphereConfig;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import redis.clients.jedis.Jedis;
23  import redis.clients.jedis.JedisPool;
24  import redis.clients.jedis.JedisPubSub;
25  import redis.clients.jedis.exceptions.JedisException;
26  
27  import java.net.URI;
28  import java.util.concurrent.atomic.AtomicBoolean;
29  
30  public class RedisUtil {
31      private static final Logger logger = LoggerFactory.getLogger(RedisBroadcaster.class);
32  
33      private static final String REDIS_AUTH = RedisBroadcaster.class.getName() + ".authorization";
34      private static final String REDIS_SERVER = RedisBroadcaster.class.getName() + ".server";
35      private static final String REDIS_SHARED_POOL = RedisBroadcaster.class.getName() + ".sharedPool";
36  
37      private Jedis jedisSubscriber;
38      private Jedis jedisPublisher;
39      private String authToken = null;
40  
41      private boolean sharedPool = false;
42      private JedisPool jedisPool;
43      private final AtmosphereConfig config;
44      private URI uri;
45      private final AtomicBoolean destroyed = new AtomicBoolean();
46      private final Callback callback;
47  
48      public RedisUtil(URI uri, AtmosphereConfig config, Callback callback) {
49          this.config = config;
50          this.callback = callback;
51          this.uri = uri;
52      }
53  
54      public String getAuth() {
55          return authToken;
56      }
57  
58      public void setAuth(String auth) {
59          authToken = auth;
60      }
61  
62      public void configure() {
63  
64          if (config.getServletConfig().getInitParameter(REDIS_AUTH) != null) {
65              authToken = config.getServletConfig().getInitParameter(REDIS_AUTH);
66          }
67  
68          if (config.getServletConfig().getInitParameter(REDIS_SERVER) != null) {
69              uri = URI.create(config.getServletConfig().getInitParameter(REDIS_SERVER));
70          } else if (uri == null) {
71              throw new NullPointerException("uri cannot be null");
72          }
73  
74          if (config.getServletConfig().getInitParameter(REDIS_SHARED_POOL) != null) {
75              sharedPool = Boolean.parseBoolean(config.getServletConfig().getInitParameter(REDIS_SHARED_POOL));
76          }
77  
78          logger.info("{} shared connection pool {}", getClass().getName(), sharedPool);
79  
80          if (sharedPool) {
81              if (config.properties().get(REDIS_SHARED_POOL) != null) {
82                  jedisPool = (JedisPool) config.properties().get(REDIS_SHARED_POOL);
83              }
84  
85              // setup is synchronized, no need to sync here as well.
86              if (jedisPool == null) {
87                  GenericObjectPool.Config gConfig = new GenericObjectPool.Config();
88                  gConfig.testOnBorrow = true;
89                  gConfig.testWhileIdle = true;
90  
91                  jedisPool = new JedisPool(gConfig, uri.getHost(), uri.getPort());
92  
93                  config.properties().put(REDIS_SHARED_POOL, jedisPool);
94              } else {
95                  disconnectSubscriber();
96              }
97          }
98  
99          // We use the pool only for publishing
100         jedisSubscriber = new Jedis(uri.getHost(), uri.getPort());
101         try {
102             jedisSubscriber.connect();
103             auth(jedisSubscriber);
104         } catch (JedisException e) {
105             logger.error("failed to connect subscriber", e);
106             disconnectSubscriber();
107         }
108 
109         jedisPublisher = sharedPool ? null : new Jedis(uri.getHost(), uri.getPort());
110         if (!sharedPool) {
111             try {
112                 jedisPublisher.connect();
113                 auth(jedisPublisher);
114             } catch (JedisException e) {
115                 logger.error("failed to connect publisher", e);
116                 disconnectPublisher();
117             }
118         }
119     }
120 
121     public synchronized void setID(String id) {
122         disconnectPublisher();
123         disconnectSubscriber();
124     }
125 
126     /**
127      * {@inheritDoc}
128      */
129     public void destroy() {
130         if (!sharedPool) {
131             Object lockingObject = getLockingObject();
132             synchronized (lockingObject) {
133                 try {
134                     disconnectPublisher();
135                     disconnectSubscriber();
136                     if (jedisPool != null) {
137                         jedisPool.destroy();
138                     }
139                 } catch (Throwable t) {
140                     logger.warn("Jedis error on close", t);
141                 } finally {
142                     config.properties().put(REDIS_SHARED_POOL, null);
143                 }
144             }
145         }
146     }
147 
148     public boolean isShared(){
149         return sharedPool;
150     }
151 
152     /**
153      * {@inheritDoc}
154      */
155     public void incomingBroadcast() {
156         logger.info("Subscribing to: {}", callback.getID());
157 
158         jedisSubscriber.subscribe(new JedisPubSub() {
159 
160             public void onMessage(String channel, String message) {
161                 callback.broadcastReceivedMessage(message);
162             }
163 
164             public void onSubscribe(String channel, int subscribedChannels) {
165                 logger.debug("onSubscribe: {}", channel);
166             }
167 
168             public void onUnsubscribe(String channel, int subscribedChannels) {
169                 logger.debug("onUnsubscribe: {}", channel);
170             }
171 
172             public void onPSubscribe(String pattern, int subscribedChannels) {
173                 logger.debug("onPSubscribe: {}", pattern);
174             }
175 
176             public void onPUnsubscribe(String pattern, int subscribedChannels) {
177                 logger.debug("onPUnsubscribe: {}", pattern);
178             }
179 
180             public void onPMessage(String pattern, String channel, String message) {
181                 logger.debug("onPMessage: pattern: {}, channel: {}, message: {}",
182                         new Object[]{pattern, channel, message});
183             }
184         }, callback.getID());
185     }
186 
187     public void outgoingBroadcast(Object message) {
188         // Marshal the message outside of the sync block.
189         String contents = message.toString();
190 
191         Object lockingObject = getLockingObject();
192         synchronized (lockingObject) {
193             if (destroyed.get()) {
194                 logger.debug("JedisPool closed. Re-opening");
195                 setID(callback.getID());
196             }
197 
198             if (sharedPool) {
199                 for (int i = 0; i < 10; ++i) {
200                     boolean valid = true;
201                     Jedis jedis = jedisPool.getResource();
202 
203                     try {
204                         auth(jedis);
205                         jedis.publish(callback.getID(), contents);
206                     } catch (JedisException e) {
207                         valid = false;
208                         logger.warn("outgoingBroadcast exception", e);
209                     } finally {
210                         if (valid) {
211                             jedisPool.returnResource(jedis);
212                         } else {
213                             jedisPool.returnBrokenResource(jedis);
214                         }
215                     }
216 
217                     if (valid) {
218                         break;
219                     }
220                 }
221             } else {
222                 try {
223                     jedisPublisher.publish(callback.getID(), contents);
224                 } catch (JedisException e) {
225                     logger.warn("outgoingBroadcast exception", e);
226                 }
227             }
228         }
229     }
230 
231     private void auth(Jedis jedis) {
232         if (authToken != null) {
233             jedis.auth(authToken);
234         }
235     }
236 
237     private void disconnectSubscriber() {
238         if (jedisSubscriber == null) return;
239 
240         synchronized (jedisSubscriber) {
241             try {
242                 jedisSubscriber.disconnect();
243             } catch (JedisException e) {
244                 logger.error("failed to disconnect subscriber", e);
245             }
246         }
247     }
248 
249     private void disconnectPublisher() {
250         if (jedisPublisher == null) return;
251 
252         synchronized (jedisPublisher) {
253             try {
254                 jedisPublisher.disconnect();
255             } catch (JedisException e) {
256                 logger.error("failed to disconnect publisher", e);
257             }
258         }
259     }
260 
261     private Object getLockingObject() {
262         return sharedPool ? jedisPool : jedisPublisher;
263     }
264 
265     public static interface Callback {
266 
267         String getID();
268 
269         void broadcastReceivedMessage(String message);
270 
271     }
