58      public final static String SOCKETIO_PACKET = SocketIOSessionManagerImpl.SocketIOProtocol.class.getName();
59      private static final Logger logger = LoggerFactory.getLogger(SocketIOAtmosphereInterceptor.class);
60      private static final int BUFFER_SIZE_DEFAULT = 8192;
61      private final SocketIOSessionManager sessionManager = new SocketIOSessionManagerImpl();;
62      /**
63       * See <a href="https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server">https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server</a><br>
64       * The timeout for the server when it should send a new heartbeat to the client. <br>
65       * In milliseconds.
66       */
67      private long heartbeatInterval = 25000;
68      /**
69       * See <a href="https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server">https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server</a><br>
70       * The timeout for the client – when it closes the connection it still has X amounts of seconds to re-open the connection. <b>This value is sent to the client after a successful handshake.</b><br>
71       * In milliseconds.
72       */
73      private long timeout = 60000;

75      /**
76       * See <a href="https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server">https://github.com/LearnBoost/socket.io/wiki/Configuring-Socket.IO#wiki-server</a><br>
77       * The timeout for the client, we should receive a heartbeat from the server within this interval. This should be greater than the heartbeat interval. <b>This value is sent to the client after a successful handshake.</b><br>
78       * In milliseconds.
79       */
80      private long heartbeatTimeout = 60000;
81      
82      private static int suspendTime = 20000;
83      private final Map<String, Transport> transports = new HashMap<String, Transport>();
84      private String availableTransports;

106     @Override
107     public Action inspect(AtmosphereResource r) {
108         final AtmosphereRequest request = r.getRequest();
109         final AtmosphereResponse response = r.getResponse();
110 
111         final AtmosphereHandler atmosphereHandler = r.getAtmosphereHandler();
112         try {
113             // find the transport
114             String path = request.getPathInfo();
115             if (path == null || path.length() == 0 || "/".equals(path)) {
116                 logger.debug("Not a SocketIO client");
117                 return Action.CONTINUE;
118             }
119 
120             if (path.startsWith("/")) {
121                 path = path.substring(1);
122             }
123 
124             String[] parts = path.split("/");
125 
126             String protocol = null;
127             String version = null;
128 
129             // find protocol's version
130             if (parts.length == 0) {
131                 logger.debug("Not a SocketIO protocol supported");
132                 return Action.CONTINUE;
133             } else if (parts.length == 1) {
134 
135                 // is protocol's version ?
136                 if (parts[0].length() == 1) {
137                     version = parts[0];
138                     // must be a digit
139                     if (!Character.isDigit(version.charAt(0))) {
140                         version = null;
141                     }
142                 } else {
143                     protocol = parts[0];
144                 }
145 
146             } else {
147                 // ex  :[1, xhr-polling, 7589995670715459]
148                 version = parts[0];
149                 protocol = parts[1];
150 
151                 // must be a digit
152                 if (!Character.isDigit(version.charAt(0))) {
153                     version = null;
154                     protocol = null;
155                 }
156 
157             }
158 
159             if (protocol == null && version == null) {
160                 logger.debug("Not a SocketIO protocol supported");
161                 return Action.CONTINUE;
162             } else if (protocol == null && version != null) {
163                 // create a session and send the available transports to the client
164                 response.setStatus(200);
165                 
166                 response.setContentType("plain/text");
167                 
168                 SocketIOSession session = getSessionManager(version).createSession((AtmosphereResourceImpl) r, atmosphereHandler);
169                 response.getWriter().print(session.getSessionId() + ":" + (heartbeatTimeout/1000) + ":" + (timeout/1000) + ":" + availableTransports);
170 
171                 return Action.CANCELLED;
172             } else if (protocol != null && version == null) {
173                 version = "0";
174             }
175 
176             final Transport transport = transports.get(protocol + "-" + version);
177             if (transport != null) {
178                 if (!SocketIOAtmosphereHandler.class.isAssignableFrom(atmosphereHandler.getClass())) {
179                     response.asyncIOWriter(new AsyncIOWriterAdapter() {
180                         @Override
181                         public AsyncIOWriter write(AtmosphereResponse r, String data) throws IOException {
182                             SocketIOSessionOutbound outbound = (SocketIOSessionOutbound)
183                                     request.getAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND);
184                             SocketIOSessionManagerImpl.SocketIOProtocol p = (SocketIOSessionManagerImpl.SocketIOProtocol)
185                                     r.request().getAttribute(SOCKETIO_PACKET);
186 
187                             String msg = p == null ? data : mapper.writeValueAsString(p.clearArgs().addArgs(data));
188 
189                             if (outbound != null) {
190                                 outbound.sendMessage(new SocketIOPacketImpl(SocketIOPacketImpl.PacketType.EVENT, msg));
191                             } else {
192                                 r.getResponse().getOutputStream().write(msg.getBytes(r.getCharacterEncoding()));
193                             }
194                             return this;
195                         }
196 
197                         @Override
198                         public AsyncIOWriter write(AtmosphereResponse r, byte[] data) throws IOException {
199                             SocketIOSessionManagerImpl.SocketIOProtocol p = (SocketIOSessionManagerImpl.SocketIOProtocol)
200                                     r.request().getAttribute(SOCKETIO_PACKET);
201                             if (p == null) {
202                                 r.getResponse().getOutputStream().write(data);
203                             } else {
204                                 write(r, new String(data, r.request().getCharacterEncoding()));
205                             }
206                             return this;
207                         }
208 
209                         @Override
210                         public AsyncIOWriter write(AtmosphereResponse r, byte[] data, int offset, int length) throws IOException {
211                             SocketIOSessionManagerImpl.SocketIOProtocol p = (SocketIOSessionManagerImpl.SocketIOProtocol)
212                                     r.request().getAttribute(SOCKETIO_PACKET);
213                             if (p == null) {
214                                 r.getResponse().getOutputStream().write(data, offset, length);
215                             } else {
216                                 write(r, new String(data, offset, length, r.request().getCharacterEncoding()));
217                             }
218                             return this;
219                         }
220 
221                         @Override
222                         public AsyncIOWriter flush(AtmosphereResponse r) throws IOException {
223                             try {
224                                 r.getResponse().getOutputStream().flush();
225                             } catch (IllegalStateException ex) {
226                                 r.getResponse().getWriter().flush();
227                             }
228                             return this;
229                         }
230 
231                         @Override
232                         public AsyncIOWriter writeError(AtmosphereResponse r, int errorCode, String message) throws IOException {
233                             ((HttpServletResponse) r.getResponse()).sendError(errorCode, message);
234                             return this;
235                         }
236                     });
237                 }
238                 transport.handle((AtmosphereResourceImpl) r, atmosphereHandler, getSessionManager(version));
239             } else {
240                 logger.error("Protocol not supported : " + protocol);
241             }
242         } catch (Exception e) {
243             logger.error("", e);
244         }
245         return Action.CONTINUE;
246     }

252     @Override
253     public void configure(AtmosphereConfig config) {
254         String s = config.getInitParameter(SOCKETIO_TRANSPORT);
255         availableTransports = s;
256 
257         if (availableTransports == null) {
258             availableTransports = "websocket,xhr-polling,jsonp-polling";
259         }
260 
261         String timeoutWebXML = config.getInitParameter(SOCKETIO_TIMEOUT);
262         if (timeoutWebXML != null) {
263             timeout = Integer.parseInt(timeoutWebXML);
264         }
265 
266         String heartbeatWebXML = config.getInitParameter(SOCKETIO_HEARTBEAT);
267         if (heartbeatWebXML != null) {
268             heartbeatInterval = Integer.parseInt(heartbeatWebXML);
269         }
270         
271         String heartbeatTimeoutWebXML = config.getInitParameter(SOCKETIO_HEARTBEAT_TIMEOUT);
272         if (heartbeatTimeoutWebXML != null) {
273             heartbeatTimeout = Integer.parseInt(heartbeatTimeoutWebXML);
274         }
275 
276         String suspendWebXML = config.getInitParameter(SOCKETIO_SUSPEND);
277         if (suspendWebXML != null) {
278             suspendTime = Integer.parseInt(suspendWebXML);
279         }
280 
281         // VERSION 1
282         WebSocketTransport websocketTransport1 = new WebSocketTransport();
283         XHRPollingTransport xhrPollingTransport1 = new XHRPollingTransport(BUFFER_SIZE_DEFAULT);
284         JSONPPollingTransport jsonpPollingTransport1 = new JSONPPollingTransport(BUFFER_SIZE_DEFAULT);
285         transports.put(websocketTransport1.getName() + "-1", websocketTransport1);
286         transports.put(xhrPollingTransport1.getName() + "-1", xhrPollingTransport1);
287         transports.put(jsonpPollingTransport1.getName() + "-1", jsonpPollingTransport1);
288 
289         sessionManager.setTimeout(timeout);
290         sessionManager.setHeartbeatInterval(heartbeatInterval);
291         sessionManager.setRequestSuspendTime(suspendTime);
292 
293     }
