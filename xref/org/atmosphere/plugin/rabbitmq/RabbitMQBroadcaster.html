16  package org.atmosphere.plugin.rabbitmq;
17  
18  import com.rabbitmq.client.AMQP;
19  import com.rabbitmq.client.Channel;
20  import com.rabbitmq.client.DefaultConsumer;
21  import com.rabbitmq.client.Envelope;
22  import com.rabbitmq.client.MessageProperties;
23  import org.atmosphere.cpr.AtmosphereConfig;
24  import org.atmosphere.cpr.Broadcaster;
25  import org.atmosphere.cpr.BroadcasterFuture;
26  import org.atmosphere.cpr.Deliver;
27  import org.atmosphere.util.SimpleBroadcaster;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.io.IOException;
32  
33  /**
34   * Simple {@link org.atmosphere.cpr.Broadcaster} implementation based on RabbitMQ
35   *
36   * @author Thibault Normand
37   * @author Jean-Francois Arcand
38   */
39  public class RabbitMQBroadcaster extends SimpleBroadcaster {
40  
41      private static final Logger logger = LoggerFactory.getLogger(RabbitMQBroadcaster.class);
42  
43      private String queueName;
44      private String consumerTag;
45      private RabbitMQConnectionFactory factory;
46      private Channel channel;
47      private String exchangeName;
48  
49      public RabbitMQBroadcaster() {}
50  
51      public Broadcaster initialize(String id, AtmosphereConfig config) {
52          super.initialize(id, config);
53          factory = RabbitMQConnectionFactory.getFactory(config);
54          channel = factory.channel();
55          exchangeName = factory.exchangeName();
56          return this;
57      }
58  
59      @Override
60      public void setID(String id) {
61          super.setID(id);
62          restartConsumer();
63      }
64  
65      @Override
66      public String getID() {
67          String id = super.getID();
68          if (id.startsWith("/*")) {
69              id = "atmosphere";
70          }
71          return id;
72      }
73  
74      @Override
75      protected void push(Deliver entry) {
76          if (destroyed.get()) {
77              return;
78          }
79  
80          outgoingBroadcast(entry.getMessage());
81      }
82  
83      public void outgoingBroadcast(Object message) {
84          try {
85              String id = getID();
86              logger.trace("Outgoing broadcast : {}", message);
87  
88              channel.basicPublish(exchangeName, id,
89                      MessageProperties.PERSISTENT_TEXT_PLAIN, message.toString().getBytes());
90  
91  
92          } catch (IOException e) {
93              logger.warn("Failed to send message over RabbitMQ", e);
94          }
95      }
96  
97      void restartConsumer() {
98          try {
99              final String id = getID();
100 
101             if (consumerTag != null) {
102                 logger.debug("Delete consumer {}", consumerTag);
103                 channel.basicCancel(consumerTag);
104                 consumerTag = null;
105             }
106 
107             if (queueName != null) {
108                 logger.debug("Delete queue {}", queueName);
109                 channel.queueUnbind(queueName, exchangeName, id);
110                 channel.queueDelete(queueName);
111                 queueName = null;
112             }
113 
114             queueName = channel.queueDeclare().getQueue();
115             channel.queueBind(queueName, exchangeName, id);
116 
117             logger.info("Create AMQP consumer on queue {}, for routing key {}", queueName, id);
118 
119             DefaultConsumer queueConsumer = new DefaultConsumer(channel) {
120                 @Override
121                 public void handleDelivery(String consumerTag,
122                                            Envelope envelope,
123                                            AMQP.BasicProperties properties,
124                                            byte[] body)
125                         throws IOException {
126 
127                     // Not for us.
128                     if (!envelope.getRoutingKey().equalsIgnoreCase(id)) return;
129 
130                     String message = new String(body);
131                     try {
132                         Object newMsg = filter(message);
133                         // if newSgw == null, that means the message has been filtered.
134                         if (newMsg != null) {
135                             deliverPush(new Deliver(newMsg, new BroadcasterFuture<Object>(newMsg), message), true);
136                         }
137                     } catch (Throwable t) {
138                         logger.error("failed to push message: " + message, t);
139                     }
140                 }
141             };
142 
143             consumerTag = channel.basicConsume(queueName, true, queueConsumer);
144             logger.info("Consumer " + consumerTag + " for queue {}, on routing key {}", queueName, id);
145 
146         } catch (Throwable ex) {
147             String msg = "Unable to initialize RabbitMQBroadcaster";
148             logger.error(msg, ex);
149             throw new IllegalStateException(msg, ex);
150         }
151     }
152 
153     @Override
154     public synchronized void releaseExternalResources() {
155         try {
156             if (channel != null && channel.isOpen()) {
157                 if (consumerTag != null) {
158                     channel.basicCancel(consumerTag);
159                 }
160             }
161         } catch (Exception ex) {
162             logger.trace("", ex);
163         }
164     }
165 }
