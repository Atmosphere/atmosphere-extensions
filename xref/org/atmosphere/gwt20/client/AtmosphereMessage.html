16  package org.atmosphere.gwt20.client;
17  
18  import java.io.Serializable;
19  
20  /**
21   * Base class for GWT deserialized messages that will be broadcasted as String through the Atmosphere Framework own component like {@link ManagedService}
22   *
23   * @author Jeanfrancois Arcand
24   */
25  public interface AtmosphereMessage<T> extends Serializable {
26  
27      public enum TYPE {STRING, BYTES}
28  
29      /**
30       * Return the {@link TYPE}
31       *
32       * @return
33       */
34      public TYPE type();
35  
36      /**
37       * Return the message of type T
38       *
39       * @return
40       */
41      public T getMessage();
42  
43      /**
44       * Set the message of type T
45       *
46       * @param message
47       */
48      public void setMessage(T message);
49  
50      /**
51       * Serialize this object in the form of an {@link String}.
52       *
53       * @return
54       */
55      public String asString();
56  
57      /**
58       * Serialize this object in the form of an byte[] array.
59       *
60       * @return
61       */
62      public byte[] asByte();
63  
64  }
