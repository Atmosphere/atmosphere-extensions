16  package org.atmosphere.plugin.rabbitmq;
17  
18  import com.rabbitmq.client.AMQP;
19  import com.rabbitmq.client.Channel;
20  import com.rabbitmq.client.DefaultConsumer;
21  import com.rabbitmq.client.Envelope;
22  import com.rabbitmq.client.MessageProperties;
23  import com.rabbitmq.client.ShutdownListener;
24  import com.rabbitmq.client.ShutdownSignalException;
25  
26  import org.atmosphere.cpr.AtmosphereConfig;
27  import org.atmosphere.cpr.Broadcaster;
28  import org.atmosphere.cpr.BroadcasterFuture;
29  import org.atmosphere.cpr.Deliver;
30  import org.atmosphere.util.SimpleBroadcaster;
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  import java.io.IOException;
35  
36  /**
37   * Simple {@link org.atmosphere.cpr.Broadcaster} implementation based on RabbitMQ
38   *
39   * @author Thibault Normand
40   * @author Jean-Francois Arcand
41   */
42  public class RabbitMQBroadcaster extends SimpleBroadcaster implements ShutdownListener {
43  
44      private static final Logger logger = LoggerFactory.getLogger(RabbitMQBroadcaster.class);
45  
46      private String queueName;
47      private String consumerTag;
48      private RabbitMQConnectionFactory factory;
49      private Channel channel;
50      private String exchangeName;
51  
52      public RabbitMQBroadcaster() {
53      }
54  
55      @Override
56      public Broadcaster initialize(String id, AtmosphereConfig config) {
57          super.initialize(id, config);
58          init(config);
59          return this;
60      }
61  
62      public void init(AtmosphereConfig config) {
63          factory = RabbitMQConnectionFactory.getFactory(config);
64          channel = factory.channel();
65          channel.addShutdownListener(this);
66  
67          exchangeName = factory.exchangeName();
68          restartConsumer();
69      }
70  
71      @Override
72      public Broadcaster initialize(String name, java.net.URI uri, AtmosphereConfig config) {
73          super.initialize(name, uri, config);
74          init(config);
75          return this;
76      }
77  
78      @Override
79      public void setID(String id) {
80          super.setID(id);
81          restartConsumer();
82      }
83  
84      @Override
85      public String getID() {
86          String id = super.getID();
87          if (id.startsWith("/*")) {
88              id = "atmosphere";
89          }
90          return id;
91      }
92  
93      @Override
94      protected void push(Deliver entry) {
95          if (destroyed.get()) {
96              return;
97          }
98  
99          outgoingBroadcast(entry.getMessage());
100     }
101 
102     public void outgoingBroadcast(Object message) {
103         try {
104             String id = getID();
105             logger.trace("Outgoing broadcast : {}", message);
106 
107             channel.basicPublish(exchangeName, id,
108                     MessageProperties.PERSISTENT_TEXT_PLAIN, message.toString().getBytes());
109 
110 
111         } catch (IOException e) {
112             logger.warn("Failed to send message over RabbitMQ", e);
113         }
114     }
115 
116     void restartConsumer() {
117         try {
118             final String id = getID();
119 
120             if (consumerTag != null) {
121                 logger.debug("Delete consumer {}", consumerTag);
122                 channel.basicCancel(consumerTag);
123                 consumerTag = null;
124             }
125 
126             if (queueName != null) {
127                 logger.debug("Delete queue {}", queueName);
128                 channel.queueUnbind(queueName, exchangeName, id);
129                 channel.queueDelete(queueName);
130                 queueName = null;
131             }
132 
133             queueName = channel.queueDeclare().getQueue();
134             channel.queueBind(queueName, exchangeName, id);
135 
136             logger.info("Create AMQP consumer on queue {}, for routing key {}", queueName, id);
137 
138             DefaultConsumer queueConsumer = new DefaultConsumer(channel) {
139                 @Override
140                 public void handleDelivery(String consumerTag,
141                                            Envelope envelope,
142                                            AMQP.BasicProperties properties,
143                                            byte[] body)
144                         throws IOException {
145 
146                     // Not for us.
147                     if (!envelope.getRoutingKey().equalsIgnoreCase(id)) return;
148 
149                     String message = new String(body);
150                     try {
151                         Object newMsg = filter(message);
152                         // if newSgw == null, that means the message has been filtered.
153                         if (newMsg != null) {
154                             deliverPush(new Deliver(newMsg, new BroadcasterFuture<Object>(newMsg), message), true);
155                         }
156                     } catch (Throwable t) {
157                         logger.error("failed to push message: " + message, t);
158                     }
159                 }
160             };
161 
162             consumerTag = channel.basicConsume(queueName, true, queueConsumer);
163             logger.info("Consumer " + consumerTag + " for queue {}, on routing key {}", queueName, id);
164 
165         } catch (Throwable ex) {
166             String msg = "Unable to initialize RabbitMQBroadcaster";
167             logger.error(msg, ex);
168             throw new IllegalStateException(msg, ex);
169         }
170     }
171 
172     @Override
173     public synchronized void releaseExternalResources() {
174         try {
175             if (channel != null && channel.isOpen()) {
176                 if (consumerTag != null) {
177                     channel.basicCancel(consumerTag);
178                 }
179             }
180         } catch (Exception ex) {
181             logger.trace("", ex);
182         }
183     }
184 
185     @Override
186     public void shutdownCompleted(ShutdownSignalException cause) {
187         this.destroy();
188     }
189 }
