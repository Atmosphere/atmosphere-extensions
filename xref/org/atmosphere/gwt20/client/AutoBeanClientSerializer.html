32  package org.atmosphere.gwt20.client;

34  import com.google.gwt.core.client.GWT;
35  import com.google.gwt.user.client.rpc.SerializationException;
36  import com.google.web.bindery.autobean.shared.AutoBean;
37  import com.google.web.bindery.autobean.shared.AutoBeanCodex;
38  import com.google.web.bindery.autobean.shared.AutoBeanFactory;
39  import com.google.web.bindery.autobean.shared.AutoBeanUtils;
40  
41  import java.util.ArrayList;
42  import java.util.HashMap;
43  import java.util.List;
44  import java.util.Map;
45  import java.util.logging.Logger;

47  /**
48   * The base class for serializers. To instantiate this class follow this example:
49   * <pre><code>
50   * <p/>
51   * {@literal @SerialTypes({ MyType1.class, MyType2.class })}
52   * public abstract class MyCometSerializer extends AtmosphereGWTSerializer {}
53   * <p/>
54   * AtmosphereGWTSerializer serializer = GWT.create(MyCometSerializer.class);
55   * AtmosphereClient client = new AtmosphereClient(url, serializer, listener);
56   * </code></pre>
57   * <p/>
58   * Where MyType1 and MyType2 are the types that your expecting to receive from the server.
59   * If you have a class hierarchy of messages that you want to send you only need to supply the base class here.
60   * <p/>
61   * For instance:
62   * <pre><code>
63   * public class Message {}
64   * <p/>
65   * public class MessageA extends Message {}
66   * <p/>
67   * public class MessageB extends Message {}
68   * <p/>
69   * {@literal @SerialTypes( Message.class )}
70   * public abstract class MyCometSerializer extends AtmosphereGWTSerializer {}
71   * <p/>
72   * </code></pre>
73   */
74  public class AutoBeanClientSerializer implements ClientSerializer {
75  
76      private static final Logger logger = Logger.getLogger(AutoBeanClientSerializer.class.getName());
77  
78      private Map<Class, AutoBeanFactory> beanFactories;
79      private AutoBeanFactory activeBeanFactory;
80      private Class<Object> activeBeanClass;
81      // buffer in order to capture split messages
82      private StringBuffer buffer = new StringBuffer(16100);
83  
84      public void registerBeanFactory(Class<AutoBeanFactory> factoryClass, Class forBean) {
85          registerBeanFactory((AutoBeanFactory) GWT.create(factoryClass), forBean);
86      }
87  
88      public void registerBeanFactory(AutoBeanFactory factory, Class forBean) {
89          if (beanFactories == null) {
90              beanFactories = new HashMap<Class, AutoBeanFactory>();
91          }
92          beanFactories.put(forBean, factory);
93          if (activeBeanFactory == null) {
94              setActiveBeanFactory(forBean);
95          }
96      }
97  
98      public void setActiveBeanFactory(Class forBean) {
99          if (beanFactories == null) {
100             throw new IllegalStateException("No bean factory available");
101         }
102         AutoBeanFactory factory = beanFactories.get(forBean);
103         if (factory == null) {
104             throw new IllegalStateException("No bean factory available");
105         }
106         activeBeanFactory = factory;
107         activeBeanClass = forBean;
108     }
109 
110     @Override
111     public Object deserialize(String raw) throws SerializationException {
112 
113         buffer.append(raw); // TODO buffer messages in case we receive a chunked message
114 
115         // split up in different parts - based on the {}
116         // this is necessary because multiple objects can be chunked in one single string
117         int brackets = 0;
118         int start = 0;
119         List<String> messages = new ArrayList<String>();
120         for (int i = 0; i < buffer.length(); i++) {
121 
122             // detect brackets
123             if (buffer.charAt(i) == '{') {
124                 ++brackets;
125             } else if (buffer.charAt(i) == '}') --brackets;
126 
127             // new message
128             if (brackets == 0) {
129                 messages.add(buffer.substring(start, i + 1));
130                 start = i + 1;
131             }
132         }
133         buffer.delete(0, start);
134 
135         // create the objects
136         List<Object> objects = new ArrayList<Object>(messages.size());
137         for (String message : messages) {
138             try {
139 
140                 logger.info("Deserialize " + message + " from " + raw);
141                 Object event = AutoBeanCodex.decode(activeBeanFactory, activeBeanClass, message).as();
142 
143                 objects.add(event);
144 
145             } catch (RuntimeException e) {
146 
147                 throw new SerializationException(e);
148 
149             }
150         }
151         return objects;
152     }
153 
154     @Override
155     public String serialize(Object message) throws SerializationException {
156         try {
157 
158             AutoBean<Object> bean = AutoBeanUtils.getAutoBean(message);
159 
160             return AutoBeanCodex.encode(bean).getPayload();
161 
162         } catch (RuntimeException e) {
163 
164             throw new SerializationException(e);
165 
166         }
167     }
