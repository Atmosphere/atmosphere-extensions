40  package org.atmosphere.guice;
41  
42  import com.google.inject.Provides;
43  import com.google.inject.servlet.ServletModule;
44  import com.sun.jersey.api.core.ExtendedUriInfo;
45  import com.sun.jersey.api.core.HttpContext;
46  import com.sun.jersey.api.core.HttpRequestContext;
47  import com.sun.jersey.api.core.HttpResponseContext;
48  import com.sun.jersey.core.util.FeaturesAndProperties;
49  import com.sun.jersey.spi.MessageBodyWorkers;
50  import com.sun.jersey.spi.container.ExceptionMapperContext;
51  import com.sun.jersey.spi.container.WebApplication;
52  import javax.ws.rs.core.HttpHeaders;
53  import javax.ws.rs.core.Request;
54  import javax.ws.rs.core.SecurityContext;
55  import javax.ws.rs.core.UriInfo;
56  import javax.ws.rs.ext.Providers;
57  
58  /**
59   * A {@link ServletModule} module that provides JAX-RS and Jersey related
60   * bindings.
61   * <p>
62   * The module has a dependency on {@link GuiceContainer}, which is required
63   * to be served in the {@link ServletModule#configure() } method.
64   * <p>
65   * The following bindings are defined:
66   * <ul>
67   * <li>{@link WebApplication}
68   * <li>{@link WebApplication}
69   * <li>{@link Providers}
70   * <li>{@link FeaturesAndProperties}
71   * <li>{@link MessageBodyWorkers}
72   * <li>{@link ExceptionMapperContext}
73   * <li>{@link HttpContext}
74   * <li>{@link UriInfo}
75   * <li>{@link ExtendedUriInfo}
76   * <li>{@link HttpRequestContext}
77   * <li>{@link HttpHeaders}
78   * <li>{@link Request}
79   * <li>{@link SecurityContext}
80   * <li>{@link HttpResponseContext}
81   * </ul>
82   * @author Paul.Sandoz@Sun.Com
83   */
84  public class AtmosphereJerseyServletModule extends ServletModule {
85  
86      @Provides
87      public WebApplication webApp(GuiceContainer guiceContainer) {
88          return guiceContainer.getWebApplication();
89      }
90  
91      @Provides
92      public Providers providers(WebApplication webApplication) {
93          return webApplication.getProviders();
94      }
95  
96      @Provides
97      public FeaturesAndProperties fearturesAndProperties(WebApplication webApplication) {
98          return webApplication.getFeaturesAndProperties();
99      }
100 
101     @Provides
102     public MessageBodyWorkers messageBodyWorkers(WebApplication webApplication) {
103         return webApplication.getMessageBodyWorkers();
104     }
105 
106     @Provides
107     public ExceptionMapperContext exceptionMapperContext(WebApplication webApplication) {
108         return webApplication.getExceptionMapperContext();
109     }
110 
111     @Provides
112     public HttpContext httpContext(WebApplication webApplication) {
113         return webApplication.getThreadLocalHttpContext();
114     }
115 
116     @Provides
117     public UriInfo uriInfo(WebApplication wa) {
118         return wa.getThreadLocalHttpContext().getUriInfo();
119     }
120 
121     @Provides
122     public ExtendedUriInfo extendedUriInfo(WebApplication wa) {
123         return wa.getThreadLocalHttpContext().getUriInfo();
124     }
125 
126     @Provides
127     public HttpRequestContext requestContext(WebApplication wa) {
128         return wa.getThreadLocalHttpContext().getRequest();
129     }
130 
131     @Provides
132     public HttpHeaders httpHeaders(WebApplication wa) {
133         return wa.getThreadLocalHttpContext().getRequest();
134     }
135 
136     @Provides
137     public Request request(WebApplication wa) {
138         return wa.getThreadLocalHttpContext().getRequest();
139     }
140 
141     @Provides
142     public SecurityContext securityContext(WebApplication wa) {
143         return wa.getThreadLocalHttpContext().getRequest();
144     }
145 
146     @Provides
147     public HttpResponseContext responseContext(WebApplication wa) {
148         return wa.getThreadLocalHttpContext().getResponse();
149     }
150 }
