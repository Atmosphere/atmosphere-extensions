16  package org.atmosphere.gwt20.managed;
17  
18  import org.atmosphere.cpr.BroadcastFilter;
19  import org.atmosphere.gwt20.client.AtmosphereMessage;
20  import org.atmosphere.gwt20.client.managed.RPCEvent;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  /**
25   * Wrap a message into an {@link org.atmosphere.gwt20.client.AtmosphereMessage}
26   *
27   * @author Jeanfrancois Arcand
28   */
29  public class AtmosphereMessageFilterEncoder implements BroadcastFilter {
30  
31      private final Logger logger = LoggerFactory.getLogger(AtmosphereMessageFilterEncoder.class);
32      private Class<? extends AtmosphereMessage<?>> messageClazz = RPCEvent.class;
33  
34      /**
35       * Wrap message inside our {@link RPCEvent}
36       *
37       * @param originalMessage the String message
38       * @param message         the String message
39       * @return an RPCEvent instance.
40       */
41      @Override
42      public BroadcastAction filter(String broadcasterId, Object originalMessage, Object message) {
43          if (!AtmosphereMessage.class.isAssignableFrom(message.getClass())) {
44              try {
45                  AtmosphereMessage<Object> m = (AtmosphereMessage<Object>) messageClazz.newInstance();
46                  m.setMessage(message);
47                  return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, m);
48              } catch (Exception e) {
49                  logger.warn("Oups. Make sure your RPCEvent implements AtmosphereMessage<Your Class>. Or remove the AtmosphereMessageInterceptor {}", e);
50                  return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
51              }
52          } else {
53              return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
54          }
55      }
56  
57      public AtmosphereMessageFilterEncoder classToEncode(Class<? extends AtmosphereMessage<?>> messageClazz) {
58          this.messageClazz = messageClazz;
59          return this;
60      }
61  }
