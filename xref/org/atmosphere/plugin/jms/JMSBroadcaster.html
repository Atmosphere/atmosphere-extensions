16  package org.atmosphere.plugin.jms;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.cpr.Broadcaster;
20  import org.atmosphere.util.AbstractBroadcasterProxy;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import javax.jms.Connection;
25  import javax.jms.ConnectionFactory;
26  import javax.jms.JMSException;
27  import javax.jms.Message;
28  import javax.jms.MessageConsumer;
29  import javax.jms.MessageListener;
30  import javax.jms.MessageProducer;
31  import javax.jms.Session;
32  import javax.jms.TextMessage;
33  import javax.jms.Topic;
34  import javax.naming.Context;
35  import javax.naming.InitialContext;
36  import java.net.URI;
37  
38  /**
39   * Simple {@link org.atmosphere.cpr.Broadcaster} implementation based on JMS
40   * <p/>
41   * The {@link ConnectionFactory} name's is jms/atmosphereFactory The
42   * {@link Topic} by constructing "BroadcasterId =
43   * {@link org.atmosphere.cpr.Broadcaster#getID}
44   *
45   * @author Jeanfrancois Arcand
46   */
47  public class JMSBroadcaster extends AbstractBroadcasterProxy {
48      private static final String JMS_TOPIC = JMSBroadcaster.class.getName() + ".topic";
49      private static final String JNDI_NAMESPACE = JMSBroadcaster.class.getName() + ".JNDINamespace";
50      private static final String JNDI_FACTORY_NAME = JMSBroadcaster.class.getName() + ".JNDIConnectionFactoryName";
51      private static final String JNDI_TOPIC = JMSBroadcaster.class.getName() + ".JNDITopic";
52      private static final Logger logger = LoggerFactory.getLogger(JMSBroadcaster.class);
53  
54      private Connection connection;
55      private Session consumerSession;
56      private Session publisherSession;
57      private Topic topic;
58      private MessageConsumer consumer;
59      private MessageProducer publisher;
60  
61      private String topicId = "atmosphere";
62      private String factoryName = "atmosphereFactory";
63      private String namespace = "jms/";
64  
65      public  JMSBroadcaster() {}
66  
67      public Broadcaster initialize(String id, URI uri, AtmosphereConfig config) {
68          super.initialize(id, uri, config);
69          setUp();
70          return this;
71      }
72  
73      private void setUp() {
74          try {
75              // For backward compatibility.
76              if (config.getInitParameter(JMS_TOPIC) != null) {
77                  topicId = config.getInitParameter(JMS_TOPIC);
78              }
79  
80              if (config.getInitParameter(JNDI_NAMESPACE) != null) {
81                  namespace = config.getInitParameter(JNDI_NAMESPACE);
82              }
83  
84              if (config.getInitParameter(JNDI_FACTORY_NAME) != null) {
85                  factoryName = config.getInitParameter(JNDI_FACTORY_NAME);
86              }
87  
88              if (config.getInitParameter(JNDI_TOPIC) != null) {
89                  topicId = config.getInitParameter(JNDI_TOPIC);
90              }
91  
92              logger.info("Looking up Connection Factory {}", namespace + factoryName);
93              Context ctx = new InitialContext();
94              ConnectionFactory connectionFactory = (ConnectionFactory) ctx.lookup(namespace + factoryName);
95  
96              logger.info("Looking up topic: {}", topicId);
97              topic = (Topic) ctx.lookup(namespace + topicId);
98  
99              connection = connectionFactory.createConnection();
100             consumerSession = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
101             publisherSession = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
102 
103             publisher = publisherSession.createProducer(topic);
104             connection.start();
105             logger.info("JMS created for topic {}", topicId);
106             // Unfortunately we need the getID() to complete the configuration
107             // But setID() is called after configure(), therefore we do the
108             // rest of the configuration in incomingBroadcast() (which is called
109             // once during configuration). We cannot do all the configuration in
110             // incomingBroadcast() though, as using bc.getAtmosphereConfig() would
111             // introduce a race condition (the configuration is loaded in a different
112             // thread).
113         } catch (Exception e) {
114             String msg = "Unable to configure JMSBroadcaster";
115             logger.error(msg, e);
116             throw new RuntimeException(msg, e);
117         }
118     }
119 
120     /**
121      * {@inheritDoc}
122      */
123     @Override
124     public void incomingBroadcast() {
125         // we setup the consumer in the setID method. No need to do it here too
126     }
127 
128     @Override
129     public void setID(String id) {
130         super.setID(id);
131         restartConsumer();
132     }
133 
134     void restartConsumer() {
135         try {
136             String id = getID();
137             if (id.startsWith("/*")) {
138                 id = "atmosphere";
139             }
140             if (consumer != null) {
141                 consumer.close();
142                 consumer = null;
143             }
144 
145             logger.info("Create JMS consumer: {}", id);
146             String selector = String.format("BroadcasterId = '%s'", id);
147 
148             consumer = consumerSession.createConsumer(topic, selector);
149             consumer.setMessageListener(new MessageListener() {
150                 @Override
151                 public void onMessage(Message msg) {
152                     try {
153                         TextMessage textMessage = (TextMessage) msg;
154                         String message = textMessage.getText();
155 
156                         if (message != null && bc != null) {
157                             broadcastReceivedMessage(message);
158                         }
159                     } catch (JMSException ex) {
160                         logger.warn("Failed to broadcast message", ex);
161                     }
162                 }
163             });
164             logger.info("Consumer created for topic {}, with filter {}",
165                     topicId, selector);
166         } catch (Throwable ex) {
167             String msg = "Unable to initialize JMSBroadcaster";
168             logger.error(msg, ex);
169             throw new IllegalStateException(msg, ex);
170         }
171     }
172 
173     /**
174      * {@inheritDoc}
175      */
176     @Override
177     public void outgoingBroadcast(Object message) {
178         try {
179             String id = getID();
180             if (id.startsWith("/*")) {
181                 id = "atmosphere";
182             }
183 
184             if (publisherSession == null) {
185                 throw new IllegalStateException("JMS Session is null");
186             }
187 
188             TextMessage textMessage = publisherSession.createTextMessage(message
189                     .toString());
190             textMessage.setStringProperty("BroadcasterId", id);
191             publisher.send(textMessage);
192         } catch (JMSException ex) {
193             logger.warn("Failed to send message over JMS", ex);
194         }
195     }
196 
197     /**
198      * Close all related JMS factory, connection, etc.
199      */
200     @Override
201     public synchronized void releaseExternalResources() {
202         try {
203             connection.close();
204             consumerSession.close();
205             publisherSession.close();
206             if (consumer != null) consumer.close();
207             if (publisher != null) publisher.close();
208         } catch (Throwable ex) {
209             logger.warn("releaseExternalResources", ex);
210         }
211     }
212 }
