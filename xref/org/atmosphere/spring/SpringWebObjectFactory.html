1   package org.atmosphere.spring;
2   
3   import org.atmosphere.cpr.AtmosphereFramework;
4   import org.atmosphere.cpr.AtmosphereObjectFactory;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   import org.springframework.context.annotation.AnnotationConfigApplicationContext;
8   import org.springframework.web.context.WebApplicationContext;
9   import org.springframework.web.context.support.WebApplicationContextUtils;
10  
11  /**
12   * An {@link AtmosphereObjectFactory} for Spring to be used in Servlet Base Java Apps.
13   *
14   * @author Aparup Banerjee
15   */
16  
17  public class SpringWebObjectFactory implements AtmosphereObjectFactory {
18  	
19  	private static final Logger logger = LoggerFactory.getLogger(SpringObjectFactory.class);
20  
21  	@Override
22  	public <T, U extends T> U newClassInstance(AtmosphereFramework framework,
23  			Class<T> classType, Class<U> classToInstantiate)
24  			throws InstantiationException, IllegalAccessException {
25  		WebApplicationContext parent = WebApplicationContextUtils.getWebApplicationContext(framework.getServletContext());
26  		AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();
27  	    context.setParent(parent);
28  	    context.register(classToInstantiate);
29  	    context.refresh();
30  		U t = context.getBean(classToInstantiate);
31  		if (t == null) {
32  			logger.info("Unable to find {}. Creating the object directly."
33  					+ classToInstantiate.getName());
34  			return classToInstantiate.newInstance();
35  		}
36  		return t;
37  	}
38  
39  	public String toString() {
40  		return "Spring Web ObjectFactory";
41  	}
42  
43  }
